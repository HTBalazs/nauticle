68  operand = std::move(op);
69  size_t type = (int)operand[3]->evaluate(0)[0];
70  kernel = std::make_shared<pmKernel>();
71  kernel->set_kernel_type(type, OP_TYPE==SAMPLE?false:true);

159 template <OPERATOR_TYPE OP_TYPE, size_t VAR, size_t K>
160 pmTensor pmSph_operator<OP_TYPE,VAR,K>::evaluate(int const& i, Eval_type eval_type/*=current*/) const {
161  if(!assigned) { pLogger::error_msgf("\"%s\" is not assigned to any particle system.\n", op_name.c_str()); }
162  pmTensor A_i = operand[0]->evaluate(i,eval_type);
163  float m_i = operand[1]->evaluate(i,eval_type)[0];
164  float rho_i = operand[2]->evaluate(i,eval_type)[0];
165  auto contribute = [&A_i, &m_i, &rho_i, &eval_type, this](pmTensor const& pos_i, pmTensor const& pos_j, int const& i, int const& j, float const& cell_size)->pmTensor{
166  pmTensor contribution;
167  if(i!=j || OP_TYPE==SAMPLE) {
168  pmTensor r_ji = pos_j-pos_i;
169  float d_ji = r_ji.norm();
170  if(d_ji < cell_size && (d_ji > 1e-6f || OP_TYPE==SAMPLE)) {
171  pmTensor A_j = operand[0]->evaluate(j,eval_type);
172  float m_j = operand[1]->evaluate(j,eval_type)[0];
173  float rho_j = operand[2]->evaluate(j,eval_type)[0];
174  float W_ij = kernel->evaluate(d_ji, cell_size);
175  contribution += this->process(A_i, A_j, rho_i, rho_j, m_i, m_j, r_ji, d_ji, W_ij);
176  }
177  }
178  return contribution;
179  };
180  return interact(i, operand, contribute);
181 }

183 pmTensor pmSph_operator<OP_TYPE,VAR,K>::evaluate(pmTensor const& pos_i, Eval_type eval_type/*=current*/) const {
184  if(!assigned) { pLogger::error_msgf("\"%s\" is not assigned to any particle system.\n", op_name.c_str()); }
185  if(OP_TYPE!=SAMPLE) { pLogger::error_msgf("Not supported for nodes out of particle system.\n", op_name.c_str()); }
186  auto contribute = [&pos_i, &eval_type, this](pmTensor const& pos_j, int const& j, float const& cell_size)->pmTensor{
187  pmTensor contribution;

189  float d_ji = r_ji.norm();
190  if(d_ji < cell_size && (d_ji > 1e-6f || OP_TYPE==SAMPLE)) {
191  pmTensor A_j = operand[0]->evaluate(j,eval_type);
192  float m_j = operand[1]->evaluate(j,eval_type)[0];
193  float rho_j = operand[2]->evaluate(j,eval_type)[0];
194  float W_ij = kernel->evaluate(d_ji, cell_size);
195  contribution += this->process(A_j, A_j, rho_j, rho_j, m_j, m_j, r_ji, d_ji, W_ij);
