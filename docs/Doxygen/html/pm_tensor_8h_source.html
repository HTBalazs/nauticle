159 inline pmTensor operator*(pmTensor const& lhs, pmTensor const& rhs) {
160  if(lhs.is_scalar() && !rhs.is_scalar()) {
161  return lhs[0]*rhs;
162  }
163  if(!lhs.is_scalar() && rhs.is_scalar()) {
164  return lhs*rhs[0];
165  }
166  if(!lhs.is_scalar() && !rhs.is_scalar()) {
167  if(lhs.get_numcols()!=rhs.get_numrows()) {
168  pLogger::error_msgf("Unable to multiply these tensors.\n");
169  return pmTensor{};
170  }
171  }
172  int imax = lhs.get_numrows();
173  int jmax = rhs.get_numcols();
174  pmTensor tensor{imax, jmax};
175  for(int i=0; i<imax; i++) {
176  for(int j=0; j<jmax; j++) {
177  float sum = 0;
178  for(int k=0; k<lhs.get_numcols(); k++) {
179  sum += lhs(i,k)*rhs(k,j);
180  }
181  tensor[i*jmax+j] = sum;
182  }
183  }
184  return tensor;
185 }

442 inline pmTensor cross(pmTensor const& lhs, pmTensor const& rhs) {
443  if(lhs.numel()!=3 || rhs.numel()!=3) {
444  pLogger::error_msgf("Cross product can be calculated for 3D vectors.\n");
445  }
446  pmTensor result{3,1,0};

448  result[1] = lhs[2]*rhs[0]-lhs[0]*rhs[2];
