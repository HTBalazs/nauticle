142 template <Ari_fn_type ARI_TYPE, size_t S>
143 pmTensor pmArithmetic_function<ARI_TYPE,S>::evaluate(int const& i, Eval_type eval_type/*=current*/) const {
144  switch(ARI_TYPE) {
145  case ABS : return abs(this->operand[0]->evaluate(i, eval_type));
146  case ACOS : return acos(this->operand[0]->evaluate(i, eval_type));
147  case ACOT : return acot(this->operand[0]->evaluate(i, eval_type));
148  case AND : return (this->operand[0]->evaluate(i, eval_type) && this->operand[1]->evaluate(i, eval_type));
149  case ASIN : return asin(this->operand[0]->evaluate(i, eval_type));
150  case ATAN : return atan(this->operand[0]->evaluate(i, eval_type));
151  case COS : return cos(this->operand[0]->evaluate(i, eval_type));
152  case COSH : return cosh(this->operand[0]->evaluate(i, eval_type));
153  case COT : return cot(this->operand[0]->evaluate(i, eval_type));
154  case COTH : return coth(this->operand[0]->evaluate(i, eval_type));
155  case CROSS : return cross(this->operand[0]->evaluate(i, eval_type),this->operand[1]->evaluate(i, eval_type));
156  case ELEM : {
157  pmTensor t1 = this->operand[0]->evaluate(i, eval_type);
158  pmTensor row = this->operand[1]->evaluate(i, eval_type);
159  pmTensor column = this->operand[2]->evaluate(i, eval_type);
160  if(!row.is_scalar() || !column.is_scalar()) { pLogger::error_msgf("Element indices must be scalars.\n"); }
161  if(row[0]>t1.get_numrows() || column[0]>t1.get_numcols()) { pLogger::error_msgf("Element indices out of bounds.\n"); }
162  return pmTensor{1,1,t1(row(0,0),column(0,0))};
163  }
164  case EXP : return exp(this->operand[0]->evaluate(i, eval_type));
165  case FLOOR : return floor(this->operand[0]->evaluate(i, eval_type));
166  case GT : return (this->operand[0]->evaluate(i, eval_type) > this->operand[1]->evaluate(i, eval_type));
167  case IF : {
168  pmTensor t2 = this->operand[0]->evaluate(i, eval_type);
169  if(!t2.is_scalar()) { pLogger::error_msgf("Logical value should be scalar.\n"); }
170  return (bool)t2[0] ? this->operand[1]->evaluate(i, eval_type) : this->operand[2]->evaluate(i, eval_type);
171  }
172  case INT1 : return this->operand[0]->evaluate(i, current) * this->operand[1]->evaluate(i, current);
173  case INT2 : return (3*this->operand[0]->evaluate(i, current)-this->operand[0]->evaluate(i, previous)) / 2 * this->operand[1]->evaluate(i, current);
174  case LOG : return log(this->operand[0]->evaluate(i, eval_type));
175  case LT : return (this->operand[0]->evaluate(i, eval_type) < this->operand[1]->evaluate(i, eval_type));
176  case MAGNITUDE : return this->operand[0]->evaluate(i, eval_type).norm();
177  case MAX : return max(this->operand[0]->evaluate(i, eval_type), this->operand[1]->evaluate(i, eval_type));
178  case MIN : return min(this->operand[0]->evaluate(i, eval_type), this->operand[1]->evaluate(i, eval_type));
179  case MOD : return mod(this->operand[0]->evaluate(i, eval_type), this->operand[1]->evaluate(i, eval_type));
180  case NOT : return !this->operand[0]->evaluate(i, eval_type);
181  case OR : return (this->operand[0]->evaluate(i, eval_type) || this->operand[1]->evaluate(i, eval_type));
182  case RAND : return pmRandom::random(this->operand[0]->evaluate(i, eval_type), this->operand[1]->evaluate(i, eval_type));
183  case SGN : return sgn(this->operand[0]->evaluate(i, eval_type));
184  case SIN : return sin(this->operand[0]->evaluate(i, eval_type));
185  case SINH : return sinh(this->operand[0]->evaluate(i, eval_type));
186  case SQRT : return sqrt(this->operand[0]->evaluate(i, eval_type));
187  case TAN : return tan(this->operand[0]->evaluate(i, eval_type));
188  case TANH : return tanh(this->operand[0]->evaluate(i, eval_type));
189  case TRACE : return this->operand[0]->evaluate(i, eval_type).trace();
190  case TRANSPOSE : return this->operand[0]->evaluate(i, eval_type).transpose();
191  case TRUNC : return trunc(this->operand[0]->evaluate(i, eval_type));
192  case XOR : return (this->operand[0]->evaluate(i, eval_type) != this->operand[1]->evaluate(i, eval_type));
193  case IDENTITY : {
194  pmTensor t3 = this->operand[0]->evaluate(i, eval_type);
195  if(!t3.is_scalar()) { pLogger::error_msgf("Not scalar in identity.\n"); }
196  return pmTensor::make_identity((int)t3[0]);
197  }
198  }
