151 void pmLink::update() {
152  sampling_nodes.clear();
153  if(psys.expired()) { return; }
154  float distance = (vertex2->get_position()-vertex1.get_position()).norm();
155  size_t num_samples = (size_t)(distance/thickness/2)+1;
156  sampling_nodes.push_back(vertex->get_position());
157  for(int i=1; i<num_samples-1; i++) {
158  pmTensor sample = vertex1->get_position()+i*(vertex2->get_position()-vertex1.get_position())/(num_samples-1);
159  sampling_nodes.push_back(sample);
160  }
161 }
