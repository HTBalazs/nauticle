81 pmTensor pmInteraction<S>::interact(int const& i, std::array<std::shared_ptr<pmExpression>, S> operand, Func_ith contribute) const {
82  std::shared_ptr<pmParticle_system> ps = psys.lock();
83  std::vector<unsigned int> const& start = ps->get_particle_space()->get_start();
84  std::vector<unsigned int> const& end = ps->get_particle_space()->get_end();
85  float cell_size = ps->get_particle_space()->get_domain().get_cell_size();
86  pmTensor pos_i = ps->get_value(i);
87  pmTensor grid_pos_i = ps->get_particle_space()->get_grid_position(pos_i);
88  pmTensor result;
89  std::vector<pmTensor> const& cell_iterator = ps->get_particle_space()->get_cell_iterator();
90  for(auto const& it:cell_iterator) {
91  pmTensor grid_pos_j{grid_pos_i+it};
92  int hash_j = ps->get_particle_space()->calculate_hash_key_from_grid_position(grid_pos_j);
93  if(start[hash_j]!=0xFFFFFFFF) {
94  for(int j=start[hash_j]; j<=end[hash_j]; j++) {
95  pmTensor pos_j = ps->get_value(j);
96  result += contribute(pos_i, pos_j, i, j, cell_size);
97  }
98  }
99  }
100  return result;
101 }
