85 template<typename T> void pmWorkspace::print_content(std::string const& title) const {
86  pLogger::titlef<LMA>(title.c_str());
87  int count=0;
88  for(auto const& it:get<T>()) {

91  pLogger::logf<YEL>(" %i) ",count);
92  it->printv();
93  pLogger::logf("\n");

100 template <typename T> std::vector<std::shared_ptr<T>> pmWorkspace::get() const {
101  std::vector<std::shared_ptr<T>> vecT;
102  for(auto const& it:definitions) {
103  if(it->is_hidden()) { continue; }
104  std::shared_ptr<T> type = std::dynamic_pointer_cast<T>(it);
105  if(type.use_count()>0) {
106  vecT.push_back(type);
107  }
108  }
109  return vecT;
110 }
