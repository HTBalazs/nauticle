67 template <typename T> std::shared_ptr<T> pmExpression_parser::analyse_expression(std::string infix, std::shared_ptr<pmWorkspace> workspace/*=std::make_shared<pmWorkspace>()*/) {
68  remove_spaces(infix);
69  if(!verify_infix(infix)) {
70  return std::shared_ptr<T>(nullptr);
71  }
72  std::vector<std::string> infix_table = build_table(infix);
73  if(!verify_table(infix_table)) {
74  return std::shared_ptr<T>(nullptr);
75  }
76  std::vector<std::string> postfix = parse(infix_table);
77  std::shared_ptr<pmExpression> expression;
78  try {
79  expression = build_expression_tree(postfix, workspace);
80  } catch(...) {
81  return std::shared_ptr<T>(nullptr);
82  }
83  return std::dynamic_pointer_cast<T>(expression);
84 }
